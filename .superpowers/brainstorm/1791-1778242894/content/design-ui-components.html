设计第 3 节：UI 组件结构

WikiChatDialog.vue 重构为纯 UI 壳层

组件职责划分

WikiChatDialog.vue（UI 壳层）
  • 模板渲染：消息气泡列表、输入区、来源栏、@弹窗
  • 事件绑定：发送、@选中、关闭、保存
  • 滚动管理：新消息自动滚到底部
  • 模板引用：textarea ref 用于 @ 检测光标位置
use-wiki-chat-session.ts（逻辑层）
  • 消息历史管理（push、filter）
  • 路由判断与 AI 调用
  • @ 检测与来源切换
  • 上下文窗口构建
  • 保存 Markdown 生成

模板结构概览

<template>
  <!-- 对话框外壳 -->
  <div class="wiki-chat-dialog">

    <!-- 1. Header: 标题 + 关闭按钮 -->
    <header>...</header>

    <!-- 2. SourceIndicator: 当前来源栏 -->
    <div class="source-bar">
      <span class="source-tag">📄 来源</span>
      <span>{{ currentSourcePage?.title ?? '自动匹配中...' }}</span>
    </div>

    <!-- 3. Messages: 消息列表 -->
    <div class="messages-area" ref="messagesRef">
      <template v-for="msg in session.messages">

        <!-- 系统消息（路由/切换提示） -->
        <div v-if="msg.role === 'system'" class="system-hint">
          {{ msg.content }}
        </div>

        <!-- 用户消息气泡 -->
        <div v-else-if="msg.role === 'user'" class="bubble-user">
          <div class="avatar">👤</div>
          <div class="bubble-content">{{ msg.content }}</div>
          <span class="timestamp">{{ formatTime(msg.timestamp) }}</span>
        </div>

        <!-- AI 消息气泡 -->
        <div v-else class="bubble-assistant">
          <div class="avatar">🤖</div>
          <div class="bubble-content">{{ msg.content }}</div>
          <div class="bubble-meta">
            <span>{{ formatTime(msg.timestamp) }}</span>
            <span class="source-badge">
              📄 {{ msg.sourcePage?.title }}
            </span>
          </div>
        </div>

      </template>
    </div>

    <!-- 4. MentionPopup: @ 候选列表 -->
    <div v-if="mentionPopupVisible" class="mention-popup">
      <div v-for="page in filteredPages"
           @click="switchSource(page)">
        📄 {{ page.title }}
      </div>
    </div>

    <!-- 5. InputArea: 输入框 + 发送按钮 -->
    <div class="input-area">
      <textarea
        ref="inputRef"
        v-model="inputText"
        @keydown.enter.prevent="sendMessage"
        placeholder="输入问题... 使用 @ 切换来源"
      />
      <button @click="sendMessage">➤</button>
    </div>

    <!-- 6. Footer: 保存整个对话按钮 -->
    <footer v-if="session.messages.length > 0">
      <button @click="$emit('save', buildSaveMarkdown())">
        保存对话
      </button>
    </footer>

  </div>
</template>
